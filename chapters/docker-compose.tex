\subsection{Beispiel-Datei}
\begin{frame}[fragile]
    \slidehead
    \vspace{-1em}
    \small
    % \begin{noindent}
    \begin{codeBlock}{minted language=yaml, title=\codeBlockTitle{docker-compose.yaml}}
        services:
            database:
                image: postgres:13.4-alpine
                environment:
                - POSTGRES_USER=hedgedoc
                - POSTGRES_PASSWORD=password
                - POSTGRES_DB=hedgedoc
                volumes:
                - database:/var/lib/postgresql/data
                restart: always
    \end{codeBlock}
    % \end{noindent}
\end{frame}

    \slidehead

            - uploads:/hedgedoc/public/uploads
